// Apps Script for the ACCENT/DIALECT experiment (experiment_accent.html).
// Setup: paste into Extensions > Apps Script of the Google Sheet, then
// Deploy > New deployment > Web app (Execute as: Me, Who has access:
// Anyone) and paste the web-app URL into CONFIG.saveUrl in
// experiment_accent.html. To update the code later WITHOUT changing the URL:
// Deploy > Manage deployments > edit (pencil) > Version: New version.
//
// Running intialSetup is optional — the script falls back to the
// spreadsheet it is bound to.

var sheetName = 'Sheet1'
var scriptProp = PropertiesService.getScriptProperties()

// One row is appended per trial, in this column order. The header row
// is created automatically on the first POST if the sheet is empty.
var HEADERS = [
  'timestamp',
  'experiment',
  'participant',
  'model',
  'trial',
  'condition_id',
  'condition',
  'group',
  'timbre',
  'source',
  'audio1',
  'audio2',
  'response',
  'same_speaker_truth',
  'rt_ms'
]

function intialSetup () {
  var activeSpreadsheet = SpreadsheetApp.getActiveSpreadsheet()
  scriptProp.setProperty('key', activeSpreadsheet.getId())
}

function doPost (e) {
  var lock = LockService.getScriptLock()
  lock.tryLock(10000)

  try {
    var key = scriptProp.getProperty('key')
    var doc = key ? SpreadsheetApp.openById(key) : SpreadsheetApp.getActiveSpreadsheet()
    var sheet = doc.getSheetByName(sheetName) || doc.getSheets()[0]

    if (sheet.getLastRow() === 0) {
      sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS])
    }

    var headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]
    var nextRow = sheet.getLastRow() + 1

    var newRow = headers.map(function(header) {
      return header === 'timestamp' ? new Date() : e.parameter[header]
    })

    sheet.getRange(nextRow, 1, 1, newRow.length).setValues([newRow])

    return ContentService
      .createTextOutput(JSON.stringify({ 'result': 'success', 'row': nextRow }))
      .setMimeType(ContentService.MimeType.JSON)
  }

  catch (err) {
    return ContentService
      .createTextOutput(JSON.stringify({ 'result': 'error', 'error': err.message || String(err) }))
      .setMimeType(ContentService.MimeType.JSON)
  }

  finally {
    lock.releaseLock()
  }
}
